def number_of_steps(n):
    counter = 0
    l = list()
    if n == 0:
        return 0, None
    while n != 1:
        l.append(n)
        if n % 2 == 1:
            n = 3 * n + 1
        else:
            n /= 2
        counter += 1
    
    l.append(1)
    return counter + 1 , l

def populate_collatz_dict(n):
    collatz_dict = dict()
    for i in xrange(1, n+1):
        steps, collatz_chain = number_of_steps(i)
        for j, val in enumerate(collatz_chain):
            if j not in collatz_dict:
                collatz_dict[val] = steps - j
    return collatz_dict

Generating it bottom up¶

Say we start with 1 and we want to generate the collatz trees from bottom up. We need to reverse the logic of building it top down.

  1. Numbers of form $3n$, $3n+2$ will always be $3n \rightarrow 6n$, and $3n+2 \rightarrow 6n+4$.
  2. NUmbers of form $3n+1$ will either be $3n+1 \rightarrow 6n+2$ or $3n+1 \rightarrow n$

def bottom_up_collatz(n):
    k = 4
    counter = 0
    collatz_graph = dict()
    q = collections.deque()
    q.clear()
    while counter < n:
        counter += 1
        if k % 3 == 1:
            q.append(2*k)
            q.append((k-1)/3)
            collatz_graph[k] = [2*k, (k-1)/3]
        else:
            q.append(2*k)
            collatz_graph[k] = [2*k]
        k = q.popleft()
    # Now empty the queue
    while q:
        if k not in collatz_graph:
            if k % 3 == 1:
                collatz_graph[k] = [2*k, (k-1)/3]
            else:
                collatz_graph[k] = [2*k]
        k = q.popleft()
    return collatz_graph
